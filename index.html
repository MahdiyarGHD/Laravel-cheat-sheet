ایمیل

ارسال ایمیل با استفاده از لاراول
پارامتر اول ، قالبی است که ما میخواهیم آن را به صورت ایمیل برای کاربر ارسال کنیم .
پارامتر دوم ارایه‌ای از اطلاعات است که به view انتخاب شده در پارامتر اول ارسال می‌شود .
پارامتر سوم که بخش مهمی در ارسال ایمیل است ، به شما این امکان را می دهد که گیرنده ها، فرستنده ها، موضوع ها و سایر جنبه های ایمیل را سفارشی کنید . در ادامه با دستورات پارامتر سوم اشنا می‌شویم . دستوراتی که در ادامه برای قرار میگیرد‌‌‌‌ ، باید در پارامتر سوم این دستور به صورت callback قرار بگیرد. 


Mail::send('email.view', $data, function($message){})

آدرس کسی که ایمیل باید برای آن ارسال شود 


$message->to('email@example.com', 'Mr. Example');

ارسال فایل ضمیمه همراه با ایمیل 


$message->attach('foo\bar.txt')
$message->embed('foo\bar.txt')

و بقیه تنظیمات ایمیل ارسالی 


$message->from('email@example.com', 'Mr. Example');
$message->sender('email@example.com', 'Mr. Example');
$message->returnPath('email@example.com');
$message->to('email@example.com', 'Mr. Example');
$message->cc('email@example.com', 'Mr. Example');
$message->bcc('email@example.com', 'Mr. Example');
$message->replyTo('email@example.com', 'Mr. Example');
$message->subject('Welcome to the Jungle');
$message->priority(2);

برای کار کردن این دستورات ، باید در فایل .env به این صورت تنظیمات را وارد کنید 


MAIL_DRIVER = smtp
MAIL_HOST = smtp.gmail.com
MAIL_PORT = 587
MAIL_USERNAME = your-gmail-username
MAIL_PASSWORD = your-application-specific-password
MAIL_ENCRYPTION = tls
کار‌ با فایل‌ها

// بررسی وجود داشتن یک آدرس
File::exists('path')

// ویرایش محتوی یک فایل ( پارامتر اول آدرس فایل و پارامتر دوم محتوی ) 
File::put('path', 'contents')

// افزودن متن به فایلی که آدرس آن در پارامتر اول قرار داده شده
File::append('path', 'data')

// پاک کردن فایل توسط آدرس فایل
File::delete('path')

// جابه‌جایی فایل از مکانی به مکان دیگر
File::move('path', 'target')

// کپی کردن فایل از مکانی به مکان دیگر
File::copy('path', 'target')

// دریافت نوع فایل با استفاده از آدرس فایل
File::type('path')

//  دریافت تمام دایرکتوری های درون یک دایرکتوری
File::directories('directory')

//  تشخیص قابل نوشتن بودن مسیر فایلی که داده شده است
File::isWritable('path')

//  دریافت اخرین تاریخ ساخت/ویرایش یک فایل
File::lastModified('path')

// دریافت تمام فایل های یک دایرکتوری بصورت آرایه
File::files('directory')

//  و بقیه دستورات کنترلی دایرکتوری ها
File::deleteDirectory('directory', $preserve = false);
File::copyDirectory('directory', 'destination', $options = null);
File::cleanDirectory('directory');
File::makeDirectory('path', $mode = 0777, $recursive = false);

// ارسال یک یادآور پسورد برای کاربر 
Password::remind($credentials, function($message, $user){})

// 'ایجاد 'مرا به خاطر بسپار
Auth::attempt($credentials, true)

// بررسی صحت یک کاربر

// لاگین کردن کاربر فقط یک بار بدون ذخیره کردن کوکی 
Auth::once($credentials)
